170  virtual int EnCode(char* strData)
171  {
172  uint32_t nOffset = 0;
173 
174  uint16_t nMsgID = NF_HTONS(munMsgID);
175  memcpy(strData + nOffset, (void*)(&nMsgID), sizeof(munMsgID));
176  nOffset += sizeof(munMsgID);
177 
178  uint32_t nPackSize = munSize + NF_HEAD_LENGTH;
179  uint32_t nSize = NF_HTONL(nPackSize);
180  memcpy(strData + nOffset, (void*)(&nSize), sizeof(munSize));
181  nOffset += sizeof(munSize);
182 
183  if (nOffset != NF_HEAD_LENGTH)
184  {
185  assert(0);
186  }
187 
188  return nOffset;
189  }

192  virtual int DeCode(const char* strData)

194  uint32_t nOffset = 0;
195 
196  uint16_t nMsgID = 0;
